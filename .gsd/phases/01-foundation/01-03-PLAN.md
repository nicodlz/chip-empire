    ```tsx
    import { motion, AnimatePresence } from 'framer-motion'
    import { formatNumber } from '../engine/resources'
    import Decimal from 'break_eternity.js'

    export interface TapData {
      id: number
      x: number
      y: number
      amount: Decimal
      emoji: string
    }

    interface Props {
      taps: TapData[]
      onComplete: (id: number) => void
    }

    export function TapFeedback({ taps, onComplete }: Props) {
      return (
        <AnimatePresence>
          {taps.map(tap => (
            <motion.div
              key={tap.id}
              initial={{ opacity: 1, y: 0, scale: 0.5 }}
              animate={{ opacity: 0, y: -100, scale: 1.2 }}
              exit={{ opacity: 0 }}
              transition={{ duration: 0.8, ease: 'easeOut' }}
              onAnimationComplete={() => onComplete(tap.id)}
              className="absolute pointer-events-none flex items-center gap-1 font-bold text-xl"
              style={{ 
                left: tap.x, 
                top: tap.y, 
                transform: 'translate(-50%, -50%)',
                textShadow: '0 0 10px var(--neon-green)',
                color: 'var(--neon-green)',
              }}
            >
              <span>+{formatNumber(tap.amount)}</span>
              <span>{tap.emoji}</span>
            </motion.div>
          ))}
        </AnimatePresence>
      )
    }
    ```
  </action>
  <verify>Numbers animate up with glow</verify>
  <done>Satisfying neon feedback on tap</done>

  <files>src/components/MineArea.tsx</files>
  <action>
    ```tsx
    import { useState, useCallback } from 'react'
    import { useGameStore } from '../store/gameStore'
    import { TapFeedback, TapData } from './TapFeedback'
    import { mineMineral } from '../engine/resources'
    import { MINERALS, TIER_1_MINERALS } from '../data/minerals'
    import { MineralId } from '../types/game'

    export function MineArea() {
      const mine = useGameStore(s => s.mine)
      const miningPower = useGameStore(s => s.miningPower)
      const [taps, setTaps] = useState<TapData[]>([])
      const [tapId, setTapId] = useState(0)
      const [currentMineral, setCurrentMineral] = useState<MineralId>('silicon')

      const handleTap = useCallback((e: React.MouseEvent | React.TouchEvent) => {
        e.preventDefault()
        
        const rect = (e.currentTarget as HTMLElement).getBoundingClientRect()
        const clientX = 'touches' in e ? e.touches[0].clientX : e.clientX
        const clientY = 'touches' in e ? e.touches[0].clientY : e.clientY
        const x = clientX - rect.left
        const y = clientY - rect.top

        const gained = mineMineral(currentMineral, miningPower)
        const def = MINERALS[currentMineral]
        
        setTaps(prev => [...prev, { id: tapId, x, y, amount: gained, emoji: def.emoji }])
        setTapId(prev => prev + 1)
        mine(currentMineral)
      }, [mine, miningPower, tapId, currentMineral])

      const removeTap = useCallback((id: number) => {
        setTaps(prev => prev.filter(t => t.id !== id))
      }, [])

      return (
        <div className="flex-1 flex flex-col items-center justify-center w-full">
          {/* Mineral selector */}
          <div className="flex gap-2 mb-6">
            {TIER_1_MINERALS.map(id => {
              const def = MINERALS[id]
              const isActive = currentMineral === id
              return (
                <button
                  key={id}
                  onClick={() => setCurrentMineral(id)}
                  className={`px-4 py-2 rounded-lg text-2xl transition-all ${
                    isActive 
                      ? 'bg-slate-700 ring-2 ring-[--neon-blue] scale-110' 
                      : 'bg-slate-800/50 hover:bg-slate-700/50'
                  }`}
                >
                  {def.emoji}
                </button>
              )
            })}
          </div>

          {/* Tap area */}
          <div 
            className="relative w-full max-w-md aspect-square flex items-center justify-center cursor-pointer select-none active:scale-95 transition-transform bg-slate-800/30 rounded-2xl border border-slate-700/50"
            onClick={handleTap}
            onTouchStart={handleTap}
          >
            <div className="text-8xl animate-pulse">
              {MINERALS[currentMineral].emoji}
            </div>
            <TapFeedback taps={taps} onComplete={removeTap} />
          </div>

          <p className="mt-4 text-slate-500 text-sm">
            Tap to mine {MINERALS[currentMineral].name}
          </p>
        </div>
      )
    }
    ```
  </action>

  <done>Can switch minerals, tap anywhere to mine</done>
